①如何取出每科成绩的前三名

第一步 建表

CREATE TABLE `test_course` ( 
  `id` varchar(10) NOT NULL, 
  `class` varchar(1000) DEFAULT NULL,
  `grade` int(100) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1;

第三步 查出每个科目的全部排名

SELECT S1.*,(SELECT COUNT(1) FROM TEST.`test_course` S2 
WHERE S1.CLASS = S2.CLASS AND S2.GRADE > S1.GRADE)+1 AS "paiming" 
FROM TEST.`test_course` S1   ORDER BY S1.CLASS,paiming;

第四步 取出每个科目的前三名

SELECT S1.*,(SELECT COUNT(1) FROM TEST.`test_course` S2 
WHERE S1.CLASS = S2.CLASS AND S2.GRADE > S1.GRADE)+1
 AS "paiming"
FROM TEST.`test_course` S1  HAVING paiming <=3 ORDER BY S1.CLASS,paiming;
